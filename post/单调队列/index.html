type dqueue []int

func (d *dqueue) Push(x int) {
	for len(*d) != 0 && (*d)[len(*d)-1] < x {
		// ç§»é™¤å‰é¢æ¯”è¯¥å€¼å°çš„æ•°
		*d = (*d)[:len(*d)-1]
	}
	*d = append(*d, x)
}

func (d *dqueue) Max() int {
	if len(*d) == 0 {
		return -1
	}
	return (*d)[0]
}

func (d *dqueue) Pop(n int) {
	if len(*d) != 0 && (*d)[0] == n {
		*d = (*d)[1:]
	}
}

func maxSlidingWindowQueue(nums []int, k int) []int {

	res := make([]int, 0)
	i, j := 0, 0
	q := make(dqueue, 0)
	for j <= len(nums) {
		if j-i < k {
			q.Push(nums[j])
		} else {
			t := nums[i]
			res = append(res, q.Max())
			q.Pop(t)

			}
			i++
		}
		j++
	}
	return res
}
